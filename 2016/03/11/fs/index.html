fs模块简单使用

node fs模块简单使用方法

首先在项目中安装fs

npm install fs

然后在文件中引入

var fs = require('fs')

1.打开文件

fs.readFile('name',function(err,data){
    //参数 name是打开文件的名称
    //参数 err 是错误信息
    //参数 data 是文件里面的内容

})

2.写入文件

fs.writeFile('name','data',function(err,callbackData){
   //参数 name是需要写入的文件名称
   //参数 data是需要写入文件的内容
   //回调函数中的参数与打开文件相同
})
